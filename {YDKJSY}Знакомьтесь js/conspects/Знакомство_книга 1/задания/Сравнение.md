const dayStart = "07:30";
const dayEnd = "17:45";
const reg = /(?<h>\d+):(?<m>\d+)/g;

function correctTime(time) {
    return time.replace(reg, (...rest) => {
        let { h, m } = rest[rest.length - 1];
        h = h.length < 2 ? `0${h}` : h;
        m = m.length < 2 ? `0${m}` : m;
        return `${h}:${m}`;
    })
}

function scheduleMeeting(startTime, durationMinutes) {
    if (correctTime(startTime) < dayStart) return false;
    if (correctTime(startTime) > dayEnd) return false;
    const HOURS = 60;
    const res = startTime.replace(reg, (...rest) => {
        let { h, m } = rest[rest.length - 1];
        h *= 1;
        m *= 1;
        if (m + durationMinutes >= HOURS) {
            let addHours = Math.floor((m + durationMinutes) / HOURS);
            let addMins = (m + durationMinutes) % HOURS;
            return `${h + addHours}:${addMins}`;
        }
        return `${h}:${m + durationMinutes}`
    })
    return correctTime(res) <= dayEnd;
}

 console.log(scheduleMeeting("07:00", 15));// false
 console.log(scheduleMeeting("7:15", 30)); // true
 console.log(scheduleMeeting("7:30", 30))// false
 console.log(scheduleMeeting("11:30", 60)); // true
 console.log(scheduleMeeting("17:00", 45)); // true
console.log(scheduleMeeting("17:27", 19))// false
