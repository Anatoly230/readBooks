function range(start, end) {
    const res = [];

    if (end < start) {
        res.push(end)
        return res
    };

    function r(e) {
        for (let st = start; st <= e; st++) {
            res.push(st);
        }
        return res;
    }
    return end ? r(end) : r;
};
